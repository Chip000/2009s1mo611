\frame{
  \frametitle{Roteamento e Alocação de Comprimentos de Onda (RWA)}
  \begin{itemize}
    \item{Redes WDM são redes nas quais múltiplos caminhos ópticos são 
      multiplexados em uma única fibra óptica}
    \item{Em redes WDM transparentes todos os caminhos ópticos permanecem 
      no meio óptico da origem ao destino}
  \end{itemize}
  \vspace{0.5cm}
  \begin{itemize}
  \item{O problema de RWA define as rotas e comprimentos de onda a serem 
    alocados para cada caminho óptico}
  \item{O número de comprimentos de onda utilizado pode ser minimizado 
    de forma a reduzir o custo total da rede}
  \end{itemize}
}

  \frametitle{Motivação}
  \begin{itemize}
  \item{A alocação de comprimentos de onda em redes WDM é um problema 
    NP-Completo}
  \item{Heuristicas são extensamente utilizadas na literatura 
    ({\color[rgb]{1,0,0} resultados aproximados})}
  \end{itemize}
  \begin{itemize}
  \item{Uma abordagem que pudesse obter uma solução 
    {\color[rgb]{1,0,0}EXATA} em um curto período de tempo é desejável}
  \end{itemize}
  \note[item]{A motivação do problema está no fato de que o problema de WA 
    em redes em malha é NP-Completo levando a literatura a considerar 
    métodos heurísticos que nem sempre garantem uma boa solução.}
  \note[item]{Devido a isso, a investigação de algoritmos para obtenção 
    de soluções exatas e eficientes faz-se necessária.}

\subsection{Complexidade Parametrizada}
\frame{
  \frametitle{Teoria da Complexidade Parametrizada}
  \begin{itemize}
  \item{É um ramo recente da Teoria da Complexidade}
  \item{Provê um arcabouço para a análise de problemas difíceis}
  \item{Remodela problemas difíceis concentrando a exponencialidade do 
    problema em um parâmetro}
  \item{Introduz a noção de Tratabilidade por Parâmetro-Fixo (FPT)}
  \end{itemize}
  \pause
  \begin{beamerboxesrounded}[center,shadow=true]{Problemas tratáveis por Parâmetro-Fixo}
    \begin{itemize}
    \item{Problemas nos quais o parâmetro é definido são chamados 
      problemas parametrizados}
    \item{Um problema parametrizado que permite um algoritmo FPT é dito 
      como sendo um problema tratável por parâmetro-fixo e pertence a 
      classe \textbf{FPT}}
    \end{itemize}
  \end{beamerboxesrounded}

\subsection{Complexidade Parametrizada da Coloração de Vértices}
\frame{
  \frametitle{Complexidade Parametrizada da Coloração de Vértices (1/4)}

  \begin{beamerboxesrounded}[center,shadow=true]{Noções Básicas}
    \begin{itemize}
    \item{Considere uma classe de grafos $F$ e um inteiro positivo $k$, 
      $F - ke$ denota a classe de grafos que pode ser obtida a partir de 
      $F$ ao se remover no máximo $k$ arestas}
    \item{$k$ é o parâmetro do problema}
    \end{itemize}
  \end{beamerboxesrounded}
  \note{Solução parametrizada para o problema de coloração de vértices.}
  \note[item]{Passo1}
  \note[item]{Definir $F - ke$ como a classe de grafos constituída pelos grafos da família $F$ obtidos através da remoção de, no máximo, $k$ arestas.}
  \note[item]{$k$ é o parâmetro a ser fixado}
}

\frame{
  \frametitle{Complexidade Parametrizada da Coloração de Vértices (2/4)}

  \begin{beamerboxesrounded}[center,shadow=true]{Teorema}
    \begin{itemize}
    \item{Seja $F$ uma classe de grafos contraível nas arestas e $T(m, n)$ 
      o tempo necessário para se calcular uma coloração ótima para um 
      grafo da classe $F$}
    \item{Uma coloração ótima para um grafo $G$ da classe $F - ke$, dado 
      um modulador de $G$, pode ser encontrado com a complexidade de 
      tempo $O(2^k max\{T(m+k,n), m+n+k\})$}
    \item{Um {\color[rgb]{1,0,0}modulador} é o conjunto de $k$ arestas 
      que, após serem adicionadas a $G$, transformam $G$ em um grafo da 
      classe $F$}
    \end{itemize}
  \end{beamerboxesrounded}
  \note[item]{Dada uma classe de grafos $F$ contraíveis nas arestas, e o 
    tempo necessário para calcular sua coloração.}
  \note[item]{A coloração ótima para um grafo da classe $F - ke$ pode ser 
    encontrada em tempo proporcional ao tempo gasto para colorir um grafo 
    da classe $F$ + um fator exponencial no parâmetro $k$.}
  \note[item]{O modulator é justamente o conjunto das arestas retiradas 
    do grafo da classe $F$, sendo o seu tamanho igual a $k$.}
}

\frame{
  \frametitle{Complexidade Parametrizada da Coloração de Vértices (3/4)}

  \begin{beamerboxesrounded}[center,shadow=true]{Algoritmo Linear para Grafos Cordais}
    \begin{itemize}
    \item{Um grafo é cordal se não possuir nenhum ciclo induzido $C_n$, 
      para $n \geq 4$}
    \item{Um modulador para um grafo $cordal - ke$ pode ser encontrado em 
      tempo $O(f(k)(m+n))$}
    \item{Uma coloração ótima de um grafo cordal pode ser encontrada em 
      tempo $O(m+n)$}
    \end{itemize}
  \end{beamerboxesrounded}
  \vspace{0.2cm}
  \pause
  Logo:
  \vspace{0.2cm}
  \begin{beamerboxesrounded}[center,shadow=true]{}
    Uma coloração ótima para um grafo $cordal - ke$ pode ser encontrada 
    com a complexidade de tempo $O(f(k)(m+n))$.

    Linear para um parâmetro fixo $k$
  \end{beamerboxesrounded}
  \note[item]{Um grafo é cordal se ele for triangulado, ou seja, não 
    contiver ciclos grandes.}
  \note[item]{Um modulator para a classe cordal pode ser encontrado em 
    tempo linear no tamanho da entrada + um fator exponencial no 
    parâmetro $k$.}
  \note[item]{A coloração ótima de um grafo cordal pode ser encontrada 
    em tempo linear.}
  \note[item]{Portanto, a coloração ótima de um grafo $chordal - ke$ 
    pode ser encontrada em tempo linear em relação ao tamanho da 
    entrada + um fator exponencial em relação ao parâmetro $k$.}
}

\section{Obtendo o grafo cordal}

\subsection{Definições e Objetivo}

\frame{
  \frametitle{Definições e Objetivo}
  As seguintes definições serão usadas:
  \begin{itemize}
  \item{$G$: grafo auxiliar}
  \item{$R$: grafo que define a rede original}
  \item{buraco: ciclo induzido com mais de 3 vértices}
  \item{demanda: rota na rede $R$ a ser encontrada}
  \item{$e$: uma aresta pertencente a $R$}
  \item{$w(e)$: custo da aresta $e$ na rede $R$}
  \item{$lower(u,v)$: custo do caminho mínimo de $u$ até $v$}
  \end{itemize}
  \vspace{1cm}
  O objetivo é construir G sempre mantendo este cordal.
}

\subsection{Teoremas}
\frame{
  \frametitle{Teoremas (1/4)}
  \begin{beamerboxesrounded}[center,shadow=true]{}
    Usaremos os seguintes teoremas para garantir que o grafo seja 
    cordal:
  \end{beamerboxesrounded}
  \begin{beamerboxesrounded}[center,shadow=true]{Teorema 1}
    Seja um grafo $G = (V, E)$ da classe cordal. Seja $x$ um novo vértice a
    ser inserido em $G$ ligando este a outros dois vértices $u$ e $v$ 
    ($u \in G$ e $v \in G$). Seja $p$ o caminho mínimo entre $u$ e $v$ em 
    $G$ e seja $G' = (V', E')$, onde $V' = V \cup \{x\}$ e 
    $E' = E \cup \{(x, u), (x, v)\}$. O único buraco, se houver um, é
    $p' = (x, u) + p + (x, v)$.
  \end{beamerboxesrounded}
  \pause{}
  \begin{beamerboxesrounded}[center,shadow=true]{Prova:}
    Como $G$ é cordal, o único possível buraco deve passar por $x$,
    usando as novas arestas $(x, u)$ e $(x, v)$.
  \end{beamerboxesrounded}
  \note[item]{Teoremas 1, 2 e 3 garantem que iremos construir um grafo cordal
    ao adicionar um novo nó no grafo auxiliar}
}

\frame{
  \frametitle{Teoremas (2/4)}
  \begin{beamerboxesrounded}[center,shadow=true]{}
    Se realmente houver um buraco em $G'$ faremos uma triangulação 
    acrescentando arestas entre $x$ e todo vértice de $p' - \{u, v\}$.
  \end{beamerboxesrounded}
  \begin{beamerboxesrounded}[center,shadow=true]{Teorema 2}
    Após a triangularização de $G'$ temos que $G'$ é cordal.
  \end{beamerboxesrounded}
  \pause{}
  \begin{beamerboxesrounded}[center,shadow=true]{Prova:}
    Pelo Teorema anterior o único buraco possível em $G'$ é 
    $p' = (x, u) + p + (x, v)$.\\
    Após a triangulação este buraco é desfeito. Para quaisquer 3 vértices
    ou mais de $p'$ juntamente com $x$ que formem um ciclo, necessáriamente
    há uma corda pois $x$ é ligado a todos os vértices de $p'$.\\
    Quaisquer 4 vértices de $G'$ menos $x$ não podem ter um buraco pois
    $G$ é cordal.
  \end{beamerboxesrounded}
}

\frame{
  \frametitle{Teoremas (3/4)}
  \begin{beamerboxesrounded}[center,shadow=true]{}
    Agora considere que iremos ligar o vértice $x$ a um outro vértice $t$ de 
    $G$.
  \end{beamerboxesrounded}
  \begin{beamerboxesrounded}[center,shadow=true]{Teorema 3}
    Seja $G' = (V', E')$ o grafo obtido após a triangulação. Seja 
    $G'' = (V', E'')$, onde $E'' = E' \cup \{(x, t), p''\}$ e 
    $p'' = (t, \ldots, z)$ é o caminho mínimo de t até algum vértice
    $z \in p' - \{x\}$. O único buraco, se houver um, em $G''$ é
    $p''' = (x, z) + p'' + (x, t)$.
  \end{beamerboxesrounded}
  \pause{}
  \begin{beamerboxesrounded}[center,shadow=true]{Prova:}
    Como $G$ e $G'$ são cordais, o único possível buraco deve usar
    a nova aresta $(x, t)$.
  \end{beamerboxesrounded}
}

\frame{
  \frametitle{Teoremas (4/4)}
  \begin{beamerboxesrounded}[center,shadow=true]{}
    Se realmente houver um buraco em $G''$ faremos uma triangulação 
    acrescentando arestas entre $x$ e todo vértice de $p''' - \{z, t\}$.
  \end{beamerboxesrounded}
  \vspace{1cm}
  \begin{beamerboxesrounded}[center,shadow=true]{Teorema 4}
    Após a triangularização de $G''$ temos que $G''$ é cordal.
  \end{beamerboxesrounded}
  \pause{}
  \vspace{1cm}
  \begin{beamerboxesrounded}[center,shadow=true]{Prova:}
    Similar ao Teorema 2.
  \end{beamerboxesrounded}
}

\subsection{Encontrando a rota mínima}
\frame{
  \frametitle{Encontrando a rota mínima}
  \begin{itemize}
  \item{A cada demanda obtemos o grafo $G$ e obtemos o conjunto de arestas
    usadas na triangulação de $G$, denominado $T$ (modulador)}
  \item{Devemos achar a rota mínima de $s$ até $t$ com parâmetro $\alpha$
    e $T$ para calcular as penalizações.}
  \item{Então a rota mínima pode ser definida como:}
  \end{itemize}
  \begin{beamerboxesrounded}[center,shadow=true]{Rota Mínima}
    \begin{equation}
      \min_{r}\big( \sum_{e \in r} w(e) + |T| \cdot \alpha \big)
    \end{equation}
  \end{beamerboxesrounded}
  \note[item]{A cada demanda garantimos que o grafo seja cordal usando os 
    Teoremas 1 e 2.}
  \note[item]{A cada aresta inserida para manter o grafo cordal é usada
    como penalização no algoritmo de branch-and-bound usado para encontrar
    a rota mínima.}
  \note[item]{A complexidade do algoritmo b&b é exponencial (isto não implica
    que o algoritmo é ruím na prática. Ex: simplex para pl.}
}

\subsection[Algoritmo Branch-and-Bound]{Algoritmo Branch-and-Bound para encontrar a rota mínima}
\frame{
  \frametitle{Algoritmo Branch-and-Bound (1/2)}
  \begin{beamerboxesrounded}[center,shadow=true]{Rota Mínima $(s, t)$}
    \begin{algorithm}[H]
      \Entrada{Vértice origem s e vértice destino t.}
      \Saida{Melhor rota.}
      \BlankLine{}
      $p \leftarrow s$
      \tcc*[l]{rota contém inicialmente s}
      $w \leftarrow 0$
      \tcc*[l]{custo inicialmente é 0}
      $Best \leftarrow \{ \infty, \emptyset \}$
      \tcc*[l]{melhor solução começa com custo $\infty$ e caminho vazio}
      $Rota(s, t, w, p)$\;
      \Retorna{$Best$}
      \label{alg:rota_minima}
    \end{algorithm}
  \end{beamerboxesrounded}
}

\frame{
  \frametitle{Algoritmo Branch-and-Bound (2/2)}
  \begin{beamerboxesrounded}[center,shadow=true]{Rota$(s, t, w, p)$}
    \begin{algorithm}[H]
      \Entrada{Vértice origem s, vértice destino t, custo atual w e 
        rota atual p.}
      \Saida{Melhor rota.}
      \BlankLine{}
      \ParaCada{$u \in V(R)$ vizinho de $s$ tal que $u \notin p$}{
        \eSe{$(u == t)$ e $((w + w(s, t) + \alpha \cdot |T|) < Best)$}{
          $Best \leftarrow \{(w + w(s, t) + \alpha \cdot |T|), p + (s, t)\}$\;
        }{
          \Se{$lower(u,t) + w + w(s, u) < Best$}{
            $Rota(u, t, w + w(s, u), p + u)$\;
          }
        }
      }
      \label{alg:rota}
    \end{algorithm}
  \end{beamerboxesrounded}
}
